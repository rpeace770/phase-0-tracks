#CLASS

class Game

attr_reader
attr_accessor :word

def secret_word(word)
	word = @word
end

def display(display, word, letter)
	p display
	correct_guess = word.index(letter)
	if correct_guess != nil
		display[correct_guess] = letter
	else
		display
	end
end

def message
	if display.include?("-")
		puts "You lose!"
	else
		puts "Congratulations! You guessed the word!"
	end
end

end
